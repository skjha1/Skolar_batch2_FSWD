### Javascript

```javascript
var str = 'Hello'; // String
var num = 10;       // Number
var bool = true;    // Boolean
var undefinedvar;   // Undefined
var nullvar = null; // Null
var sym = Symbol("foo");  //  Symbol


var arr = [1,2,3]; // Array 
var obj = {'name': "Harpreet", age: 21}; // Object
function greet() { console.log("hello!"); } // Function 
var date = new Date(); // Date 


console.log("String: ", str);
console.log("Number: ", num);
console.log("Boolean: ", bool);
console.log("Undefined", undefinedvar);
console.log("Null : ", nullvar);
console.log("Symbol: ", sym.toString()); // The symbol needs to be converted in string
console.log("Array : ", arr);
console.log("Object : ", obj);
console.log("Function: ", greet.toString()); // print the function declaration
console.log("Date  : ", date);
console.log("\n Calling the function: ")
greet(); // Call the function




```
